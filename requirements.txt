db.courses.aggregate([
    {
        $lookup: {
            from: "students",
            localField: "students_id",
            foreignField: "_id",
            as: "students_details"
        }
    },
    {
        $project: {
            course_name: 1,
            "students_details.name": 1,
            "students_details.age": 1
        }
    }
])
